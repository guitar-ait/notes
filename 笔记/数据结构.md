```C++
//顺序表基本操作的实现
#include<iostream>
using namespace std;
//预定义常量和类型
#define TRUE 1
#define FALSE 0
#define OK 1
#define ERROR 0
#define INFEASIBLE -1
#define OVERFLOW -2

#define MAXSIZE 100

typedef int Status;
typedef char ElemType;

//定义顺序表类型
typedef struct {
	ElemType *elem;//char类型的数组 
	int length;
}SqList; 

int main() {			
	return 0;
}

//构造空的的顺序表L 
Status initList_Sq(SqList &L) {
//分配空间 
L.elem = new ElemType[MAXSIZE];
//存储分配失败
if(!L.elem) exit(OVERFLOW);
//空表长度为0 
L.length = 0;
return OK; 
}

//销毁线性表L
void DestroyList(SqList &L) {
	if(L.elem) delete L.elem;
} 

//清空线性表
void ClearList(SqList &L) {
	L.length = 0;
} 

//求线性表的长度
int GetLength(SqList &L) {
	return(L.length);
} 

//判断线性表L是否为空 
int IsEmpty(SqList &L) {
	if(L.length == 0) return 1;
	else return 0;
}

//顺序表的取值
int GetElem(SqList &L, int i, ElemType &e) {
	//判断i值是否合理
	if(i < 1 || i > L.length) return ERROR;
	e = L.elem[i - 1];
	return OK; 
} 

//顺序表的查找
int LocateElem(SqList &L, ElemType e) {
	for(int i = 0; i < L.length; i++) 
		if(L.elem[i] == e) return i + 1;
	else return 0; 
	return 0;	
} 

//顺序表的插入
Status ListInsert_Sq(SqList &L, int i, ElemType e) {	
//判断i值是否合法
if(i < 1 || i > L.length + 1) return ERROR;
//判断是否有足够空间
if(L.length == MAXSIZE) return ERROR;
//腾位置
for(int j = L.length - 1; j >= i - 1; j--) {
	L.elem[j + 1] = L.elem[j];
}//如果i == L.length + 1,则不满足循环条件，此时不用腾位置，直接插入	
//插入 
L.elem[i - 1] = e;
//表长加1
L.length++;
return OK; 
}

//删除元素
Status ListDlete_Sq(SqList &L, int i) {
	//判断i值是否合法
	if(i < 1 || i > L.length) return ERROR;
	//元素前移，覆盖删除
	for(int j = i; j < L.length - 1; j++) {
		L.elem[j - 1] = L.elem[j];
	} 
	L.length++;
	return OK;
} 

```
